\section*{Consulta 1 -- Posologia (Simples) \hfill 1/20}

\noindent\textbf{Pergunta:} \textit{Qual a dose de dipirona para adulto?}

\vspace{2mm}
\begin{framed}
\noindent\textbf{Resposta do FarmaIA:}\\[2pt]
A dose recomendada de dipirona para adultos é de 500\,mg a 1\,g
por dose (1 a 2 comprimidos de 500\,mg), podendo ser administrada
até 4 vezes ao dia. A dose máxima diária é de 4\,g. Os comprimidos
devem ser ingeridos com líquido, sem mastigar.
\end{framed}

\begin{framed}
\noindent\textbf{Fonte (trecho da bula):}\\[2pt]
\small Posologia: Adultos e adolescentes acima de 15 anos: 500\,mg a
1\,g por administração, podendo ser administrada até 4 vezes ao dia.
Dose máxima diária: 4\,g. Os comprimidos devem ser ingeridos inteiros
com quantidade suficiente de líquido.
\end{framed}

\section*{Consulta 2 -- Posologia (Simples) \hfill 2/20}

\noindent\textbf{Pergunta:} \textit{Como tomar amoxicilina 500\,mg?}

\vspace{2mm}
\begin{framed}
\noindent\textbf{Resposta do FarmaIA:}\\[2pt]
A amoxicilina 500\,mg deve ser tomada por via oral, 1 cápsula a cada
8 horas (3 vezes ao dia), com ou sem alimentos. O tratamento
geralmente dura de 7 a 10 dias. É importante completar todo o
tratamento mesmo que os sintomas melhorem antes.
\end{framed}

\begin{framed}
\noindent\textbf{Fonte (trecho da bula):}\\[2pt]
\small Posologia: Adultos: 500\,mg a cada 8 horas, por via oral.
A duração do tratamento é de 7 a 10 dias, devendo ser mantido por
pelo menos 48 a 72 horas após o desaparecimento dos sintomas. Pode
ser administrado independentemente das refeições.
\end{framed}

\section*{Consulta 3 -- Posologia (Simples) \hfill 3/20}

\noindent\textbf{Pergunta:} \textit{Qual a posologia de ibuprofeno
para crianças de 20\,kg?}

\vspace{2mm}
\begin{framed}
\noindent\textbf{Resposta do FarmaIA:}\\[2pt]
Para crianças de 20\,kg, a dose recomendada de ibuprofeno é de 100
a 200\,mg por dose (1 a 2\,ml da suspensão 50\,mg/ml), a cada 6 a
8 horas. A dose máxima diária é de 30\,mg/kg. Não ultrapassar
600\,mg por dia.
\end{framed}

\begin{framed}
\noindent\textbf{Fonte (trecho da bula):}\\[2pt]
\small Posologia pediátrica: 5 a 10\,mg/kg/dose, a cada 6 a 8
horas. Dose máxima diária: 30\,mg/kg. Crianças de 16 a 20\,kg
(4--6 anos): 100\,mg (2\,ml). Crianças de 20 a 25\,kg (6--8 anos):
150\,mg (3\,ml). Suspensão oral 50\,mg/ml.
\end{framed}
